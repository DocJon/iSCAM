405 void OperatingModel::calcSurveyCatchability()
406 {
407  int i,k,f,g,h,ig;
408  double di;
409 
410  for(int kk = 1; kk <= nItNobs; kk++ )
411  {
412  // | Vulnerable numbers-at-age to survey
413  dmatrix V(1,n_it_nobs(kk),sage,nage);
414  V.initialize();
415 
416  for(int ii = 1; ii <= n_it_nobs(kk); ii++ )
417  {
418  i = d3_survey_data(kk)(ii)(1); // year
419  k = d3_survey_data(kk)(ii)(3); // gear
420  f = d3_survey_data(kk)(ii)(4); // area
421  g = d3_survey_data(kk)(ii)(5); // stock
422  h = d3_survey_data(kk)(ii)(6); // sex
423  di = d3_survey_data(kk)(ii)(8); // timing of survey
424 
425  dvector na(sage,nage);
426  dvector sa(sage,nage);
427  dvector va(sage,nage);
428  dvector wa(sage,nage);
429  dvector ma(sage,nage);
430  na.initialize();
431  for( h = 1; h <= nsex; h++ )
432  {
433  ig = pntr_ags(f,g,h);
434  va = exp( d4_log_sel(k)(ig)(i) );
435  sa = mfexp( -d3_Zt(ig)(i)*di );
436  na = elem_prod(d3_Nt(ig)(i),sa);
437  wa = d3_wt_avg(ig)(i);
438  ma = d3_wt_mat(ig)(i);
439  switch(n_survey_type(kk))
440  {
441  case 1:
442  V(ii) += elem_prod(na,va);
443  break;
444 
445  case 2:
446  V(ii) += elem_prod(elem_prod(na,va),wa);
447  break;
448 
449  case 3:
450  V(ii) += elem_prod(elem_prod(na,va),ma);
451  break;
452  }
453  }
454  }
455  dvector it = trans(d3_survey_data(kk))(2); //relative abundance index
456  dvector wt = trans(d3_survey_data(kk))(7); //relative weight (multiplier)
457  wt = wt / sum(wt);
458  dvector t1 = rowsum(V);
459  dvector zt = log(it) - log(t1);
460  double zbar = zt * wt;
461  m_survey_q(kk) = exp(zbar);
462 
463  } // end of nItNobs
464 
465 }

660 void OperatingModel::implementFisheries(const int& iyr)
661 {
662  /*
663  Author Steve Martell
664  Notes:
665  - This is going to be a bit tricky due to the number of factors at play.
666  - Essentially this routine must calculate the F-at-age for each gear that
667  is given a TAC > 0. Also concerned with bycatch fisheries (the effort
668  in bycatch fisheries is likely to be independent of halibut stock size).
669 
670  - For the directed fishery, the TAC value includes wastage, so must also
671  keep track of landed fish, U32 discards, and lost gear. Use the same model
672  that is described by Lew Coggins, and the work I did with Bill Pine where
673  there is a joint probability of capture * (probability of retention +
674  (1-probability of retention)*discard mortality) that is a function of size-
675  at-age and the legal size limit.
676 
677  - Aug 27, for now keep it simple
678 
679  - Sep 4, update the catch data array for writing to simulated data file.
680 
681  - Sep 10, 2013. Changed the logic in this routine. Implemented as follows:
682  - 1) Apportion m_dTac by area (f) from each stock (g)
683  - 2) Loop over each area and allocate catch in area (f) to gear (k),
684  - 3) Assemble arguments for BaranovCatchEquation class
685  -> catch by gear,
686  -> natural mortality rate by sex(row) and age(col)
687  -> Selectivity array sex,gear,age
688  -> Numbers by sex(row) and age (col)
689  -> Weight by sex(row) and age (col)
690  - 4) Calculate Fishing Mortality rates using class BaranovCatchEquation
691  - 5) Calculate Total mortality rate (d3_Zt)
692  */
693  /*
694  Given a tac in year iyr, figure out what the F is.
695  getFishingMortality in Baranov.h arguments are:
696  - ct for each fleet
697  - Natural mortality by age
698  - selectivity for each fleet
699  - numbers -at-age
700  - [weight -at-age if catch is in biomass units]
701  */
702 
703  int f,g,h,k;
704  int ig,kk,hh;
705 
706 
707  dvector tac(1,nArea);
708  dvector ct(1,nFleet);
709  dmatrix ma(1,nSex,nSage,nNage);
710  dmatrix na(1,nSex,nSage,nNage);
711  dmatrix wa(1,nSex,nSage,nNage);
712  d3_array d_Va(1,nSex,1,nFleet,nSage,nNage);
713  dmatrix d_alloc(1,nArea,1,nFleet); // Allocation for each fleet in each area.
714  tac.initialize();
715  na.initialize();
716  for( f = 1; f <= nArea; f++ )
717  {
718  // -1) Apportion catch to each area (f) from each stock (g)
719  for( g = 1; g <= nStock; g++ )
720  {
721  tac(f) += m_dTac(g);
722  }
723 
724  // -2) Allocate catch in each area (f) to gear (k).
725  // -[ ] TODO: Will need to add area specific rules here.
726  // -[ ] TODO: add implementation error here.
727  for( k = 1; k <= nFleet; k++ )
728  {
729  d_alloc(f,k) = dAllocation(k);
730  ct(k) = d_alloc(f,k)*tac(f);
731  }
732 
733  // -3) Assemble arguments for BarnovCatchEquation class.
734  // [ ] TODO: allow for time-varying M in future
735  // [ ] TODO: allow for Selectivity to change in future.
736  // [ ]
737  m_dFt.allocate(1,nStock,1,nFleet);
738  dmatrix _hCt(1,nSex,1,nFleet);
739  BaranovCatchEquation cBCE;
740  for( g = 1; g <= nStock; g++ )
741  {
742  for( h = 1; h <= nSex; h++ )
743  {
744  ig = pntr_ags(f,g,h);
745  ma(h) = d3_Mt(ig)(nNyr);
746  for( k = 1; k <= nFleet; k++ )
747  {
748  kk = nFleetIndex(k);
749  d_Va(h)(k) = exp(d4_log_sel(kk)(ig)(nNyr));
750  }
751  na(h)+= d3_Nt(ig)(iyr);
752  wa(h) = m_d3_wt_avg(ig)(iyr);
753  }
754 
755  // 4) - calculate fishing mortality rate based on baranov catch eqn.
756  // Potential issue here if nStock > 1, what wa ma vector should be used?
757  dvector ft = cBCE.getFishingMortality(ct,ma,&d_Va,na,wa,_hCt);
758  cout<<"ft = "<<ft<<endl;
759  cout<<"hCt = "<<_hCt<<endl;
760  m_dFt(g) = ft;
761 
762  // -4) Fill catch data array
763  for( k = 1; k <= nFleet; k++ )
764  {
765  if( ft(k)>0 )
766  {
767  kk = nFleetIndex(k);
768  // determine if the fleet has sex-specific or aggregated catches
769  hh = m_catch_sex_composition(k);
770  int nn = hh>0?hh:1;
771  for(h=1; h<=nn; h++)
772  {

980 void OperatingModel::calcStockRecruitment()
981 {
982  int f,g,h,j;
983  int ig;
984  double dt = d_iscamCntrl(13); // get this from cntrl(13) in the control file.
985  double phib;

993  m_dSbo.allocate(1,nStock);
994 
995 
996  for( g = 1; g <= nStock; g++ )
997  {
998  // Survivorship
999  lx.initialize();
1000  lw.initialize();
1001  lx(nSage) = 1.0;
1002  lw(nSage) = 1.0;
1003  phib = 0;
1004  for( f = 1; f <= nArea; f++ )
1005  {
1006  for( h = 1; h <= nSex; h++ )
1007  {
1008  ig = pntr_ags(f,g,h);
1009  for(j=nSage; j<=nNage; j++)
1010  {
1011  ma(j) = mean(trans(d3_Mt(ig))(j));
1012  fa(j) = mean(trans(d3_wt_mat(ig))(j));
1013 
1014  if(j>nSage)
1015  {
1016  lx(j) = lx(j-1) * mfexp(-ma(j-1));
1017  }
1018  lw(j) = lx(j) * mfexp(-ma(j)*dt);
1019  }
1020  lx(nNage) /= 1.0 - mfexp(-ma(nNage));
1021  lw(nNage) /= 1.0 - mfexp(-ma(nNage));
1022

1025  }

1028  m_so(g) = m_kappa(g)/phib;
1029  m_dSbo(g) = dRo(g) * phib;
