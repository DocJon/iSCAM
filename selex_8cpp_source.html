215 void Selex::selcoeff( const dmatrix& theta, const ivector& blk, dmatrix& log_sel)
216 {
217  /*
218  Selectivity coefficients for each age class, where it is assumed that the
219  last two age classes have the same selectivity coefficient.
220  */
221  int i,j,k;
222  int cntr = 1;
223  int r1 = log_sel.rowmin();
224  int r2 = log_sel.rowmax();
225  int c1 = log_sel.colmin();
226  int c2 = log_sel.colmax();
227 
228  log_sel.initialize();
229  k = 0;
230  for( i = r1; i <= r2; i++ )
231  {
232  if( i == blk(cntr))
233  {
234  k++;
235  if(cntr < blk.indexmax()) cntr++;
236  }
237  for( j = c1; j < c2; j++ )
238  {
239  log_sel(i,j) = theta(k)(j-c1+1);
240  }
241  log_sel(i,c2) = log_sel(i,c2-1);
242  log_sel(i) -= log( mean(mfexp(log_sel(i))) );
243  }
244 
245 }

308 static double approx1(const double& v, const dvector& x, const dvector& y)
309 {
310  /* Approximate y(v), given (x,y)[i], i = 0,..,n-1 */
311  int i, j, ij;
312 
313  //if(!n) return R_NaN;
314 
315  i = x.indexmin();
316  j = x.indexmax() - 1;
317 
318  /* handle out-of-domain points */
319  if(v < x[i]) return min(y);
320  if(v > x[j]) return max(y);
321 
322  /* find the correct interval by bisection */
323  while(i < j - 1)
324  { /* x[i] <= v <= x[j] */
325  ij = (i + j)/2; /* i+1 <= ij <= j-1 */
326  if(v < x[ij]) j = ij;
327  else i = ij;
328  /* still i < j */
329  }
330  /* provably have i == j-1 */
331 
332  /* interpolation */
333  if(v == x[j]) return y[j];
334  if(v == x[i]) return y[i];
335  /* impossible: if(x[j] == x[i]) return y[i]; */
336 
337  /* linear */
338  return y[i] + (y[j] - y[i]) * ((v - x[i])/(x[j] - x[i]));
339 }/* approx1() */
